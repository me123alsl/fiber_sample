Installation

Server Node

  1. Install Script (Master Node 공통 설치)
# Root User 
swapoff -a  #(/etc/fstab swap comment out)
apt-get upgrade -y 
apt-get dist-upgrade -y 
apt-get update -y
systemctl stop ufw && ufw disable && iptables -F
curl -sfL https://get.rke2.io | INSTALL_RKE2_TYPE="server" sh -
systemctl enable rke2-server.service

# Use Sudo
sudo swapoff -a  #(/etc/fstab swap comment out)
sudo apt-get upgrade -y 
sudo apt-get dist-upgrade -y 
sudo apt-get update -y
sudo sh -c 'systemctl stop ufw && ufw disable && iptables -F'
sudo sh -c 'curl -sfL https://get.rke2.io | INSTALL_RKE2_TYPE="server" sh -'
# sudo sh -c 'curl -sfL https://get.rke2.io | INSTALL_RKE2_TYPE="server" INSTALL_RKE2_VERSION="v1.22.6+rke2r1" sh -'
sudo sh -c 'systemctl enable rke2-server.service'
  1. CNI (Calico) Setting
# Root User
mkdir -p /etc/rancher/rke2
vi /etc/rancher/rke2/config.yaml

# Use Sudo
sudo mkdir -p /etc/rancher/rke2
sudo sh -c 'cat<<EOF > /etc/rancher/rke2/config.yaml
cni: "calico"
EOF'

sudo cat /etc/rancher/rke2

# sudo vi /etc/rancher/rke2/config.yaml
# ↓ Write Option 
# cni: "calico"
  1. Run Service RKE2-Server (1st Only)
# Root User
systemctl start rke2-server.service

# Use Sudo
sudo sh -c 'systemctl start rke2-server.service'
  1. Set Kubeconfig 
# Root User
mkdir ~/.kube/
cp /etc/rancher/rke2/rke2.yaml ~/.kube/config
export PATH=$PATH:/var/lib/rancher/rke2/bin/
echo 'export PATH=/usr/local/bin:/var/lib/rancher/rke2/bin:$PATH' >> ~/.bashrc
echo 'source <(kubectl completion bash)' >>~/.bashrc
echo 'alias k=kubectl' >>~/.bashrc
echo 'complete -F __start_kubectl k' >>~/.bashrc

# Use Sudo
sudo mkdir ~/.kube/
sudo cp /etc/rancher/rke2/rke2.yaml /home/rancher/.kube/config
export PATH=$PATH:/var/lib/rancher/rke2/bin/
sudo echo 'export PATH=/usr/local/bin:/var/lib/rancher/rke2/bin:$PATH' >> ~/.bashrc
sudo echo 'source <(kubectl completion bash)' >>~/.bashrc
sudo echo 'alias k=kubectl' >>~/.bashrc
sudo echo 'complete -F __start_kubectl k' >>~/.bashrc

sudo sh -c 'chmod -R 755 /home/rancher/.kube'
  1. Check Running CNI(Calico)
kubectl get ns calico-system
kubectl get deploy -n calico-system
kubectl get ds -n calico-system
  1. Copy RKE Server Token (1st Only)
sudo cat /var/lib/rancher/rke2/server/node-token
# Example
# K1050efa12b1c9c7b9e8d5d2f41ad7b4dfa98035b89bc0739679f338b222c681143::server:7bc6df135e949cb687b3a3ed5f578ca7
  1. Joining a node (2nd, 3rd) to the first node
sudo mkdir -p /etc/rancher/rke2
sudo vi /etc/rancher/rke2/config.yaml

# edit configure file
# server: 1st Node IP
# token: Token in [6]
server: https://192.168.0.8:9345
token: [1번노드 토큰]
  1. Run Service RKE2-Server (2nd, 3rd)
# Root User
systemctl start rke2-server.service

# Use Sudo
sudo sh -c 'systemctl start rke2-server.service'

Worker Node

  1. Install Script
# Root User
curl -sfL https://get.rke2.io | INSTALL_RKE2_TYPE="agent" sh -
mkdir -p /etc/rancher/rke2/
vi /etc/rancher/rke2/config.yaml
# server: https://<server>:9345
# token: ( Server Token )
# node-name : (선택사항)

# Use Sudo
sudo sh -c 'curl -sfL https://get.rke2.io | INSTALL_RKE2_TYPE="agent" sh -'
sudo mkdir -p /etc/rancher/rke2/
sudo vi /etc/rancher/rke2/config.yaml
#server: https://<server>:9345
#token: ( Server Token )
  1. Start RKE2 (Agent) Service
# Root User
systemctl enable rke2-agent.service
systemctl start rke2-agent.service
journalctl -u rke2-agent -f

# Use Sudo
sudo sh -c 'systemctl enable rke2-agent.service'
sudo sh -c 'systemctl start rke2-agent.service'
journalctl -u rke2-agent -f

Setup Cert-Manager

kubectl apply -f https://github.com/jetstack/cert-manager/releases/download/v1.5.4/cert-manager.yaml
kubectl -n cert-manager rollout status deploy/cert-manager
kubectl -n cert-manager rollout status deploy/cert-manager-webhook 
kubectl get pods --namespace cert-manager

Setup Helm

curl -fsSL -o get_helm.sh https://raw.githubusercontent.com/helm/helm/master/scripts/get-helm-3
chmod 700 get_helm.sh
./get_helm.sh

Setup Rancher UI

kubectl create namespace cattle-system
helm repo add rancher-stable https://releases.rancher.com/server-charts/stable
helm repo update
helm search repo rancher-stable

# Edit Replicas Count (Node Count만큼)
helm install rancher rancher-stable/rancher --namespace cattle-system --set hostname=192.168.0.8.nip.io --version=2.6.5 --set bootstrapPassword=123123123123 --set replicas=3
helm install rancher rancher-stable/rancher \
--namespace cattle-system \
--set hostname=192.168.0.8.nip.io \
--version=2.6.5 \
--set replicas=3

# Reset BootstrapPassword
kubectl -n cattle-system exec $(kubectl -n cattle-system get pods -l app=rancher | grep '1/1' | head -1 | awk '{ print $1 }') -- reset-password
helm upgrade --install rancher rancher-stable/rancher \
--namespace cattle-system \
--set hostname=10.138.0.24 \
--version=2.6.5 \

--set replicas=3
